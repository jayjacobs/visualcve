# sleep is one of:  T/F for always sleeping or not, or an integer, for "sleep under these remaining"
ratelimit <- function(header) {
  tibble(limit = as.numeric(header$`x-ratelimit-limit`),
         remaining = as.numeric(header$`x-ratelimit-remaining`), 
         reset = as.numeric(header$`x-ratelimit-reset`)) %>%
    mutate(now = parse_date_time(header$date, orders = "%a, %d %b %Y HMS", tz="GMT"),
           reset_dt = as_datetime(reset, tz="UTC")) %>% 
    # mutate(reset_dt = if_else(reset_dt < now, add_with_rollback(reset_dt, hours(1)), reset_dt)) %>%
    mutate(secdiff = as.numeric(difftime(reset_dt, now, units="secs")))
}

github_api <- function(path, config, sleep = Inf, logger = log4r::logger(), method="GET", ...) {
  url <- modify_url("https://api.github.com", path = path)
  resp <- NA
  if(method == "GET") {
    resp <- httr::GET(url, config, ...)
  } else if (method == "PUT") {
    debug(logger, "attempting a PUT request")
    resp <- httr::PUT(url, config, ...)
  }

  expect_resp <- ifelse("Accept" %in% names(config$headers), as.vector(config$headers['Accept']), "application/json")
  expect_resp <- ifelse(expect_resp == "application/vnd.github.cloak-preview",
                        "application/json", expect_resp)
  if (httr::http_type(resp) != expect_resp) {
    warn(logger, paste("API did not return expected response type:", expect_resp))
    warn(logger, paste("API sent back:", httr::http_type(resp)))
    # stop(paste("API did not return expected response type:", expect_resp), call. = FALSE)
  }
  if(status_code(resp) != 200) {
    error(logger, sprintf("GitHub API request failed [%s] <%s>", status_code(resp), url))
    # stop(sprintf("GitHub API request failed [%s]\n<%s>", status_code(resp), url), call. = FALSE)
    return(NULL)
  }
  parsed <- httr::content(resp, "text")
  if(httr::http_type(resp) == "application/json") {
    parsed <- jsonlite::fromJSON(httr::content(resp, "text", encoding = "UTF-8"), simplifyVector = FALSE)
  }
  curlimits <- ratelimit(resp$headers)
  if(curlimits$remaining <= 5) {
    sleeptime <- curlimits$secdiff * 1.05 # add a buffer just to be safe when we're close to the limit
    info(logger, paste0("Almost at reset, setting to remaining time:", round(sleeptime, 3)))
  } else {
    sleeptime <- (curlimits$secdiff / curlimits$remaining) * 0.92  # trying to be more aggressive
  }
  if(is.nan(sleeptime) | !is.finite(sleeptime)) sleeptime <- 1
  if(sleeptime < 0) sleeptime <- 0

  if(sleeptime > 3600) {
    warn(logger, "weird Inf Sleep catch triggered:")
    warn(logger, paste0("  secdiff: ", round(curlimits$secdiff, 3)))
    warn(logger, paste0("  remaining: ", round(curlimits$remaining, 3)))
    warn(logger, paste0("  sleeptime is.finite: ", is.finite(sleeptime)))
    sleeptime <- curlimits$secdiff * 1.05 # add 5% just to be safe on the last one
    warn(logger, paste0("  sleeptime is padded secdiff: ", round(sleeptime, 3)))
  }

  sleep_logical <- FALSE
  if(is.logical(sleep)) {
    sleep_logical <- sleep
  } else if (is.integer(sleep)) {
    sleep_logical <- curlimits$remaining <= sleep
  }


  if(sleep_logical) {
    debug(logger, paste0("  ", curlimits$limit, "/", curlimits$remaining, ", reset: ", curlimits$reset_dt,
                         ", sleeptime: ", round(sleeptime, 3)))
    Sys.sleep(sleeptime)
  } else {
    debug(logger, "  not sleeping")
  }

  structure(
    list(
      content = parsed,
      path = path,
      response = resp
    ),
    class = "github_api"
  )
}

gh_files <- function(repo, branch="master", config, sleep, logger = log4r::logger()) {
  curpath <- paste0("/repos/", repo, "/branches/", branch)
  cat(glue::glue("gh_files path: {curpath}\n"))
  cur_base <- list()
  try(cur_base <- github_api(curpath, config=config, sleep=sleep, logger=logger))
  rez <- tibble()                 
  if(length(cur_base) > 0) { 
      cat(glue::glue("gh_files path: {curpath}\n"))

    master_tree_sha <- cur_base$content$commit$commit$tree$sha
    curpath <- paste0("/repos/", repo, "/git/trees/", master_tree_sha)
    try(cur_base <- github_api(curpath, config=config, sleep=sleep, query=list(recursive=1), logger=logger))
    rez <- map_dfr(cur_base$content$tree, function(x) {
      enframe(unlist(x)) %>% spread(name, value)
    })
  } 
  rez
} 

gh_commit_file <- function(repo, path, msg, file, config, sleep,  logger = log4r::logger()) {
  # https://docs.github.com/en/rest/reference/repos#create-or-update-file-contents
  # set up repo/path to cleanly combine
  localpath <- gsub("^/+", "", path)
  repo <- gsub("/$+", "", repo)
  repo <- gsub("^/+", "", repo)
  curpath <- paste0("/repos/", repo, "/contents/", localpath)
  levellog(logger, "INFO", paste("file commit:", curpath))
      
  rawtxt <- readBin(file, "raw", file.info(file)[1, "size"])
  levellog(logger, "INFO", paste(length(as.integer(rawtxt)), "bytes,",
                                  "file commit:", curpath))
  bodylist <- list(message = msg, content = rawtxt)
  levellog(logger, "INFO", paste("getting existing file list"))
  curfiles <- gh_files(repo, config=config, sleep=sleep, logger=logger)
  cursha <- curfiles %>% filter(path == localpath) %>% slice(1) %>% pull(sha)
  if(length(cursha) > 0) {
    levellog(logger, "INFO", paste("attempting to overwrite existing file:", localpath))
    bodylist$sha = cursha
  } else {
    levellog(logger, "INFO", paste("attempting to create new file:", localpath))
  }
  body <- toJSON(bodylist, raw = "base64", auto_unbox=TRUE)

  try(cur_base <- github_api(curpath, config=config, sleep=sleep, logger=logger,
                             method="PUT", body = body))
  cur_base
}
